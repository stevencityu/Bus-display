// ── 修改後的警告邏輯：確保圖案能正確顯示 ── xhrGet('https://data.weather.gov.hk/weatherAPI/opendata/weather.php?dataType=warnsum&lang=tc', function(err, warn) { if (err || !warn) return; var badges = document.getElementById('warning-badges'); badges.innerHTML = ''; var tickerWarns = []; // 遍歷回傳的 API 物件 for (var key in warn) { var wObj = warn[key]; // 確保不是 CANCEL 狀態 if (wObj && wObj.actionCode !== 'CANCEL') { var label = WARN_LABEL[key] || key; // 嘗試直接使用天文台圖片網址 var iconUrl = WARN_ICON[key]; if (iconUrl) { var img = document.createElement('img'); img.className = 'warn-icon-img'; img.src = iconUrl; img.onerror = function() { // 如果圖片載入失敗，自動回退顯示文字標籤 this.style.display = 'none'; var fallback = document.createElement('span'); fallback.className = 'warn-badge ' + (WARN_CSS[key] || 'wT3'); fallback.textContent = label; badges.appendChild(fallback); }; badges.appendChild(img); } else { // 沒有圖片時顯示文字 var css = WARN_CSS[key] || 'wT3'; var b = document.createElement('span'); b.className = 'warn-badge ' + css; b.textContent = label; badges.appendChild(b); } tickerWarns.push(label); } } // 更新跑馬燈文字 if (tickerWarns.length) { document.getElementById('ticker-inner').textContent = '⚠️ 本港正生效：' + tickerWarns.join(' │ ') + ' │ 請留意天氣變化'; } });
